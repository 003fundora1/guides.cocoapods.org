            272 def download_and_extract_tarball(id)
273   File.open(tmp_path, "w+") do |tmpfile|
274     open tarball_url_for(id) do |archive|
275       tmpfile.write Zlib::GzipReader.new(archive).read
276     end
277 
278     system "tar xf #{tmpfile.path} -C #{target_path} --strip-components 1"
