    170 def self.common_build_settings(type, platform, deployment_target = nil, target_product_type = nil)
171   if target_product_type == Constants::PRODUCT_TYPE_UTI[:bundle]
172     build_settings = {
173       'PRODUCT_NAME' => '$(TARGET_NAME)',
174       'WRAPPER_EXTENSION' => 'bundle',
175       'SKIP_INSTALL' => 'YES'
176     }
177 
178     if platform == :osx
179       build_settings['COMBINE_HIDPI_IMAGES'] = 'YES'
180       build_settings['SDKROOT'] = 'macosx'
181     else
182       build_settings['SDKROOT'] = 'iphoneos'
183     end
184     build_settings
185   else
186     common_settings = Constants::COMMON_BUILD_SETTINGS
187     settings = deep_dup(common_settings[:all])
188     settings.merge!(deep_dup(common_settings[type]))
189     settings.merge!(deep_dup(common_settings[platform]))
190     settings.merge!(deep_dup(common_settings[[platform, type]]))
191     if deployment_target
192       if platform == :ios
193         settings['IPHONEOS_DEPLOYMENT_TARGET'] = deployment_target
194       elsif platform == :osx
195         settings['MACOSX_DEPLOYMENT_TARGET'] = deployment_target
196       end
197     end
198     settings
199   end
200 end

    209 def self.deep_dup(object)
210   case object
211   when Hash
212     new_hash = {}
213     object.each do |key, value|
214       new_hash[key] = deep_dup(value)
215     end
216     new_hash
217   when Array
218     object.map { |value| deep_dup(value) }
219   else
220     object.dup
221   end
222 end
